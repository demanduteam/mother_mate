مراحل استقرار در Supabase و وب‌سایت GitHub Pages

با کپی کردن این کدها، وب‌سایت شما به صورت کاملا رایگان آنلاین و متصل به بانک اطلاعاتی دائمی خواهد شد.

۱ گام اول: فعال‌سازی Supabase

یک پروژه در Supabase.com بسازید، سپس به بخش SQL Editor بروید، کد زیر را کپی کرده و اجرا (Run) کنید تا کل جدول‌های موردنیاز ایجاد شوند: 

-- Table 1: Users registered by Admin
CREATE TABLE public.users (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name TEXT NOT NULL,
  phone VARCHAR(20) UNIQUE NOT NULL,
  password TEXT NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Table 2: Audio files
CREATE TABLE public.audios (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  title TEXT NOT NULL,
  description TEXT,
  url TEXT NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Table 3: Questionnaire Questions
CREATE TABLE public.questions (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  title TEXT NOT NULL,
  type VARCHAR(20) NOT NULL, -- 'rating', 'choice', 'text'
  options TEXT[], -- Array of choices
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Table 4: Survey Responses
CREATE TABLE public.responses (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_phone VARCHAR(20) NOT NULL,
  answers JSONB NOT NULL,
  listened_audios TEXT[],
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
